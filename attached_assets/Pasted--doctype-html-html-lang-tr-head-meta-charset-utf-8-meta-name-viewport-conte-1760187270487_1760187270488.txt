    (function(){
       const images = [
  'img/bir.jpeg',
  'img/iki.jpeg',
  'img/üc.jpeg',
  'img/dort.jpeg',
  'img/bes.jpeg',
  'img/alti.jpeg',
  'img/yedi.jpeg'
];



      const root = document.getElementById('heroSlider');
      const dotsWrap = document.getElementById('heroDots');
      if(!root || !dotsWrap) return;

      const slides = images.map((src, idx)=>{
        const s = document.createElement('div');
        s.className = 'slide' + (idx===0 ? ' active' : '');
        const img = document.createElement('img');
        img.loading = 'lazy';
        img.src = src;
        img.alt = 'Proje görseli ' + (idx+1);
        s.appendChild(img);
        root.appendChild(s);
        return s;
      });

      const dots = images.map((_,i)=>{
        const b = document.createElement('button');
        b.setAttribute('aria-label','Slayt ' + (i+1));
        if(i===0) b.classList.add('active');
        b.addEventListener('click', ()=>go(i, true));
        dotsWrap.appendChild(b);
        return b;
      });

      let i = 0; let timer; const D = 4000;
      function go(n, user=false){
        slides[i].classList.remove('active');
        dots[i].classList.remove('active');
        i = (n + slides.length) % slides.length;
        slides[i].classList.add('active');
        dots[i].classList.add('active');
        if(user){ reset(); }
      }
      function next(){ go(i+1); }
      function reset(){ clearInterval(timer); timer = setInterval(next, D); }
      root.addEventListener('mouseenter', ()=>clearInterval(timer));
      root.addEventListener('mouseleave', reset);
      reset();
    })();
  </script>
